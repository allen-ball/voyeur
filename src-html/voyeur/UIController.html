023import ball.spring.AbstractController;
024import ball.upnp.ssdp.SSDPMessage;
025import java.net.URI;
026import java.util.List;
027import java.util.Map;
028import java.util.concurrent.ConcurrentSkipListMap;
029import lombok.NoArgsConstructor;
030import lombok.ToString;
031import lombok.extern.log4j.Log4j2;
032import org.springframework.beans.factory.annotation.Autowired;
033import org.springframework.stereotype.Controller;
034import org.springframework.ui.Model;
035import org.springframework.web.bind.annotation.ModelAttribute;
036import org.springframework.web.bind.annotation.RequestMapping;
037
038import static java.util.stream.Collectors.groupingBy;
039import static java.util.stream.Collectors.mapping;
040import static java.util.stream.Collectors.toList;
041
042/**
043 * UI {@link Controller} implementation.
044 *
045 * {@injected.fields}
046 *
047 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
048 * @version $Revision$
049 */
050@Controller
051@NoArgsConstructor @ToString @Log4j2
052public class UIController extends AbstractController {
053    @Autowired private SSDP ssdp = null;
054    @Autowired private NetworkInterfaces interfaces = null;
055    @Autowired private ARPCache arp = null;
056    @Autowired private Nmap nmap = null;
057
058    @ModelAttribute("upnp")
059    public Map<URI,List<URI>> upnp() {
060        Map<URI,List<URI>> map =
061            ssdp().values()
062            .stream()
063            .map(SSDP.Value::getSSDPMessage)
064            .collect(groupingBy(SSDPMessage::getLocation,
065                                ConcurrentSkipListMap::new,
066                                mapping(SSDPMessage::getUSN, toList())));
067
068        return map;
069    }
070
071    @ModelAttribute("ssdp")
072    public SSDP ssdp() { return ssdp; }
073
074    @ModelAttribute("interfaces")
075    public NetworkInterfaces interfaces() { return interfaces; }
076
077    @ModelAttribute("arp")
078    public ARPCache arp() { return arp; }
079
080    @ModelAttribute("nmap")
081    public Nmap nmap() { return nmap; }
082
083    @RequestMapping(value = {
084                        "/",
085                        "/upnp/devices", "/upnp/ssdp",
086                        "/network/interfaces", "/network/arp", "/network/nmap"
087                    })
088    public String root(Model model) { return getViewName(); }
089
090    @RequestMapping(value = { "/index", "/index.htm", "/index.html" })
091    public String index() { return "redirect:/"; }
092}
