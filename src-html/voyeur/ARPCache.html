023import ball.annotation.CompileTimeCheck;
024import java.io.BufferedReader;
025import java.io.InputStream;
026import java.io.InputStreamReader;
027import java.net.InetAddress;
028import java.nio.file.Files;
029import java.nio.file.Path;
030import java.nio.file.Paths;
031import java.util.regex.Matcher;
032import java.util.regex.Pattern;
033import lombok.NoArgsConstructor;
034import lombok.extern.log4j.Log4j2;
035import org.springframework.boot.context.event.ApplicationReadyEvent;
036import org.springframework.context.event.EventListener;
037import org.springframework.scheduling.annotation.Scheduled;
038import org.springframework.stereotype.Service;
039import voyeur.types.HardwareAddress;
040
041import static java.lang.ProcessBuilder.Redirect.PIPE;
042import static java.nio.charset.StandardCharsets.UTF_8;
043import static java.util.stream.Collectors.toMap;
044
045/**
046 * {@link java.net.InetAddress} to {@link HardwareAddress}
047 * {@link java.util.Map}.
048 *
049 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
050 * @version $Revision$
051 */
052@Service
053@NoArgsConstructor @Log4j2
054public class ARPCache extends InetAddressMap<HardwareAddress> {
055    private static final long serialVersionUID = 8171059827964411042L;
056
057    private static final Path PATH = Paths.get("/proc/net/arp");
058
059    private static final ProcessBuilder BUILDER =
060        new ProcessBuilder("arp", "-an")
061        .inheritIO()
062        .redirectOutput(PIPE);
063
064    @CompileTimeCheck
065    private static final Pattern PATTERN =
066        Pattern.compile("(?i)^(?<host>.+)"
067                        + " [(](?<inet>[\\p{Digit}.]+)[)]"
068                        + " at (?<mac>[\\p{XDigit}:]+) .*$");
069
070    /** @serial */ private boolean disabled = false;
071
072    @EventListener(ApplicationReadyEvent.class)
073    @Scheduled(fixedDelay = 60 * 1000)
074    public void update() {
075        if (! isDisabled()) {
076            try {
077                ARPCache map =
078                    Files.exists(PATH) ? parse(PATH) : parse(BUILDER);
079
080                if (map != null) {
081                    putAll(map);
082                    keySet().retainAll(map.keySet());
083                }
084            } catch (Exception exception) {
085                log.error(exception.getMessage(), exception);
086            }
087        }
088    }
089
090    public boolean isDisabled() { return disabled; }
091
092    private ARPCache parse(Path path) throws Exception {
093        ARPCache map =
094            Files.lines(path, UTF_8)
095            .skip(1)
096            .map(t -> t.split("[\\p{Space}]+"))
097            .collect(toMap(k -> getInetAddress(k[0]),
098                           v -> new HardwareAddress(v[3]),
099                           (t, u) -> t, ARPCache::new));
100
101        return map;
102    }
103
104    private ARPCache parse(ProcessBuilder builder) throws Exception {
105        ARPCache map = null;
106
107        try {
108            Process process = builder.start();
109
110            try (InputStream in = process.getInputStream()) {
111                map =
112                    new BufferedReader(new InputStreamReader(in, UTF_8))
113                    .lines()
114                    .map(PATTERN::matcher)
115                    .filter(Matcher::matches)
116                    .collect(toMap(k -> getInetAddress(k.group("inet")),
117                                   v -> new HardwareAddress(v.group("mac")),
118                                   (t, u) -> t, ARPCache::new));
119            }
120
121            disabled = (process.waitFor() != 0);
122        } catch (Exception exception) {
123            disabled = true;
124        }
125
126
127        if (disabled) {
128            log.warn("arp command is not available");
129        }
130
131        return map;
132    }
133
134    private InetAddress getInetAddress(String string) {
135        InetAddress address = null;
136
137        try {
138            address = InetAddress.getByName(string);
139        } catch (RuntimeException exception) {
140            throw exception;
141        } catch (Exception exception) {
142            throw new IllegalStateException(exception);
143        }
144
145        return address;
146    }
147}
